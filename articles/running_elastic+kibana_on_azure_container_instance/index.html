Running Elastic+Kibana on Azure Container Instance

300 Words | Approximately 2 Minutes Read | Last Modified on September 8, 2019

  • Start from azure-quickstart-templates\201-aci-wordpress (https://github.com/Azure/azure-quickstart-templates.git)
"containers": [
  {
    "name": "elasticsearch",
    "properties": {
      "image": "elasticsearch:7.3.1",
      "resources": {
        "requests": {
          "cpu": "[variables('cpuCores')]",
          "memoryInGb": "[variables('memoryInGb')]"
        }
      }
    }
  },
  {
    "name": "kibana",
    "properties": {
      "image": "kibana:7.3.1",
      "ports": [
        {
          "protocol": "Tcp",
          "port": 5601
        }
      ],
      "environmentVariables": [
        {
          "name": "ELASTICSEARCH_HOSTS",
          "value": "http://127.0.0.1:9200"
        }
      ],
      "resources": {
        "requests": {
          "cpu": "[variables('cpuCores')]",
          "memoryInGb": "[variables('memoryInGb')]"
        }
      }
    }
  }
]
